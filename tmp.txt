            const maxTicks = 6; // Exactly 6 movements per candle before closing

            // Creates the basic DOM structure for a candlestick
            function createCandleDOM() {
                const wrapper = document.createElement('div');
                wrapper.className = 'relative h-full w-1.5 md:w-2.5 flex justify-center shrink-0';

                const wick = document.createElement('div');
                const body = document.createElement('div');

                wrapper.appendChild(wick);
                wrapper.appendChild(body);
                return { wrapper, wick, body };
            }

            // Applies visual styling and physical dimensions based on price values
            function applyStyles(elements, isBullish, open, current, high, low, isLive) {
                const colorClassBody = isBullish ? 'bg-emerald-400' : 'bg-rose-400';
                const colorClassWick = isBullish ? 'bg-emerald-500/50' : 'bg-rose-500/50';
                const shadowClass = isBullish ? 'shadow-[0_0_12px_rgba(52,211,153,0.25)]' : 'shadow-[0_0_12px_rgba(251,113,133,0.25)]';
                
                // Live candles have a smooth transition on the body, closed ones are static
                const opacity = isLive ? 'opacity-100' : 'opacity-50';
                const transition = isLive ? 'transition-all duration-[600ms] ease-out' : '';

                // min-h-[1px] ensures Doji visibility without breaking percentage-based alignment
                elements.body.className = `absolute w-full rounded-[1px] z-10 ${colorClassBody} ${opacity} ${shadowClass} min-h-[1px] ${transition}`;
                
                // Wicks do NOT use transition, snapping instantly so they never visually displace from the high/low
                elements.wick.className = `absolute w-px ${colorClassWick}`;

                const bodyTop = Math.min(open, current);
                const bodyHeight = Math.abs(open - current); // Exact calculation

                elements.body.style.top = `${bodyTop}%`;
                elements.body.style.height = `${bodyHeight}%`;

                elements.wick.style.top = `${high}%`;
                elements.wick.style.height = `${low - high}%`;
            }

            // Finishes the old candle and sets up a new one
            function startNewCandle() {
                if (liveCandle) {
                    // Freeze the current live candle's final state
                    const isBullish = currentY <= openY;
                    applyStyles(liveCandle, isBullish, openY, currentY, highY, lowY, false);
                }

                // Reset trackers for the newly opening candle
                openY = currentY;
                highY = currentY;
                lowY = currentY;
                ticks = 0;

                liveCandle = createCandleDOM();
                chart.appendChild(liveCandle.wrapper);

                // Clean up oldest candles so the browser doesn't slow down
                if (chart.children.length > 80) {
                    chart.removeChild(chart.firstChild);
                }

                updateLiveVisuals();
            }

            // Triggers the CSS updates visually for the live candle
            function updateLiveVisuals() {
                const isBullish = currentY <= openY;
                applyStyles(liveCandle, isBullish, openY, currentY, highY, lowY, true);

                // Update floating horizontal price line indicator
                priceLine.style.top = `${currentY}%`;
                if (isBullish) {
                    priceLine.classList.replace('border-rose-500/50', 'border-emerald-500/50');
                    priceDot.classList.replace('bg-rose-500', 'bg-emerald-500');
                    priceDot.classList.replace('shadow-[0_0_12px_rgba(244,63,94,0.9)]', 'shadow-[0_0_12px_rgba(16,185,129,0.9)]');
                } else {

                    priceDot.classList.replace('bg-emerald-500', 'bg-rose-500');
                    priceDot.classList.replace('shadow-[0_0_12px_rgba(16,185,129,0.9)]', 'shadow-[0_0_12px_rgba(244,63,94,0.9)]');
                }
            }

            // Executes a single price movement 
            function tick() {
                if (ticks >= maxTicks) {
                    startNewCandle();
                    // Espera 1 segundo exacto antes de empezar a moverse la nueva vela
                    setTimeout(tick, 1000);

                }

                // Simulate market volatility with a random walk delta
                let delta = (Math.random() - 0.5) * 8; 
                
                // Soft limits to keep price generally centered on screen
                if (currentY < 20) delta += 3;
                if (currentY > 80) delta -= 3;

                currentY += delta;

                // Expand highs/lows limits based on new current price
                if (currentY < highY) highY = currentY; // Remember: smaller Y is higher vertically
                if (currentY > lowY) lowY = currentY;

                updateLiveVisuals();
                ticks++;

                // Ritmo normal de movimiento
                setTimeout(tick, 800);
            }

                chart.appendChild(cDOM.wrapper);
                pastY = pClose;
